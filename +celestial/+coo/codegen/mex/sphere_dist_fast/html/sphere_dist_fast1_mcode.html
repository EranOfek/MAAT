1function [Dist,PA]=sphere_dist_fast(RA_1,Dec_1,RA_2,Dec_2)
 2%--------------------------------------------------------------------------
 3% sphere_dist_fast function                                          ephem
 4% Description: Calculate the angular distance between two points on the
 5%              celestial sphere. See sphere_dist.m (and built in distance.m)
 6%                for a more general function. This function is ~10 time
 7%              faster than sphere_dist.m, but it works only with radians
 8%                and calculate only the distance.
 9% Input  : - Matrix of logitudes for the first point [radian].
10%          - Matrix of latitudes for the first point [radian].
11%          - Matrix of logitudes for the second point [radian].
12%          - Matrix of latitudes for the second point [radian].
13% Output : - Matrix of distances between points [radian].
14%          - Matrix of position angles between points [radian].
15% Tested : Matlab 2011b
16%     By : Eran O. Ofek                    Feb 2013
17%    URL : http://weizmann.ac.il/home/eofek/matlab/
18% See also: sphere_dist.m, sphere_dist_cosd.m (and built in distance.m).
19% Example: D=celestial.coo.sphere_dist_fast(RA1,Dec1,RA2,Dec2);
20% Reliable: 1
21%--------------------------------------------------------------------------
22
23single RA_1, RA_2, Dec_1, Dec_2
24Dist = acos(sin(Dec_1).*sin(Dec_2) + cos(Dec_1).*cos(Dec_2).*cos(RA_1-RA_2));
25%Dist = acos(sin(Dec_1).*sin(Dec_2) + sqrt(1-sin(Dec_1).^2).*sqrt(1-sin(Dec_2).^2).*cos(RA_1-RA_2));  % this is more accurate
26
27
28if (nargout>1)
29   dRA = RA_1-RA_2;
30   SinPA = sin(dRA).*cos(Dec_2)./sin(Dist);
31   CosPA = (sin(Dec_2).*cos(Dec_1) - cos(Dec_2).*sin(Dec_1).*cos(dRA))./sin(Dist);
32  
33   PA    = atan2(SinPA,CosPA);
34   %PA(PA<0) = 2.*pi + PA(PA<0);
35   
36   I     = find(PA<0);
37   PA(I) = 2.*pi + PA(I);
38
39end
